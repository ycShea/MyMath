graph TD
    %% Define Node Styles
    classDef user fill:#e1f5fe,stroke:#01579b,stroke-width:2px;
    classDef logic fill:#fff9c4,stroke:#fbc02d,stroke-width:2px;
    classDef function fill:#e8f5e9,stroke:#2e7d32,stroke-width:2px;
    classDef scene fill:#fce4ec,stroke:#c2185b,stroke-width:2px;
    classDef animation fill:#eeeeee,stroke:#616161,stroke-width:2px;

    %% --- THE PASSIVE WAIT STATE ---
    WAIT[<b>State:</b> Waiting for User Action]
    WAIT:::animation

    %% --- USER INTERACTION ---
    CLICK{<b>User Click</b><br/>Button Event}:::user
    WAIT --> CLICK

    %% --- LOGIC FLOW: REVEAL STEP ---
    subgraph revealStepLogic [Function: revealStep()]
        direction TB
        CHECK_PART{<b>Logic Check</b><br/>Is it Part A or B?}:::logic
        CHECK_FIG{<b>Logic Check</b><br/>Evaluate<br/>'currentFigure' state}:::logic
    end
    CLICK -- Calls revealStep() --> CHECK_PART

    CHECK_PART -- Part A --> CHECK_FIG
    CHECK_PART -- Part B --> CHECK_FIG

    %% --- DRAWING TRIGGER ---
    HT_A[<b>Drawing Trigger</b><br/>Calls handleToggle('A')]:::logic
    HT_B[<b>Drawing Trigger</b><br/>Calls handleToggle('B')]:::logic

    %% Decision based on current state and required state
    CHECK_FIG -- "currentFigure is 'null' or 'B'" --> HT_A
    CHECK_FIG -- "currentFigure is 'null' or 'A'" --> HT_B
    CHECK_FIG -- "Figure already matches" --> REVEAL_STEP[Reveal Solution Text<br/>& Highlight Triangle]:::function

    %% --- SCENE UPDATE ---
    subgraph handleToggleLogic [Function: handleToggle()]
        direction TB
        CLEAR[<b>ClearScene()</b><br/>Remove old objects<br/>& labels]:::scene
        DRAW_A[<b>drawFigureA()</b><br/>Define Part A Geometry]:::function
        DRAW_B[<b>drawFigureB()</b><br/>Define Part B Geometry]:::function
        ADD_A[<b>Scene Update</b><br/>scene.add(Base Figure A)]:::scene
        ADD_B[<b>Scene Update</b><br/>scene.add(Base Figure B)]:::scene
        UPDATE_CAM[<b>resetCamera()</b><br/>Set Camera View]:::function
    end

    HT_A --> CLEAR
    HT_B --> CLEAR
    CLEAR --> DRAW_A
    CLEAR --> DRAW_B
    DRAW_A --> ADD_A
    DRAW_B --> ADD_B
    ADD_A --> UPDATE_CAM
    ADD_B --> UPDATE_CAM

    %% Back to original process
    UPDATE_CAM --> REVEAL_STEP

    %% --- VISUAL CHANGE ---
    subgraph animateLoop [Continous Loop: animate()]
        ANIM[<b>animate()</b><br/>Runs every frame]:::animation
        PROJ[Project HTML Labels]:::function
        RENDER[<b>Visual Change</b><br/>Render Scene Graph]:::scene
    end

    REVEAL_STEP --> ANIM
    ANIM --> PROJ
    PROJ --> RENDER
    RENDER -- Next Frame --> ANIM
